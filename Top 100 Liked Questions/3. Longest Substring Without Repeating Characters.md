 
## Problem Description
```
Given a string, find the length of the longest substring without repeating characters.
```

## Ideas
Brute Force
Sliding Window + Hash Table

## Solutions
```
class Solution {
public:
    bool isunique(string s, int start, int end){
    unordered_set<char> set;
        for (int i = start; i < end; ++i){
            char temp = s[i];
            auto it = set.find(temp);
            if (it!=set.end()){
                return false;
            }
            set.insert(temp);
        }
        return true;
    }
    
    int lengthOfLongestSubstring(string s) {
        int best = 0;
        for (int i=0; i < s.length(); ++i){
            for (int j=i+1; j <= s.length(); ++j){
                if (isunique(s,i,j)){
                    best = max (best, j-i);
                }
            }   
        }      
        return best;    
    }
};
// Brute Force: check all the unique substring and get the longest length 
// Time Complexity: O(N^3)
// Time Limit Exceeded!
```

```
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int n = s.length();
        unordered_set<char> set;
        int best = 0;
        int i=0,j=0;
        while ( i<n && j<n ){
            auto it = set.find(s[i]);
            if ( it == set.end() ){
                set.insert(s[i]);
                ++i;
                best = max(best,i-j);
            }
            else{
                set.erase(s[j]);
                ++j;
            }
        }
        return best;  
    }   
};
// Sliding Window : two indexes i&j, put s[i] into set if not in [j,i) and increase i, 
                    erase s[j] from set if s[i] is in [j,i) and increase j, record the largest value by i-j
// Time Complexity: O(N)
```
